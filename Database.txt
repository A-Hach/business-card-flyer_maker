create database PrintShop;
use PrintShop;
create table Users(Id int auto_increment primary key,Email varchar(50),Pass varchar(50),TypeUser varchar(50));
create table Produits(Id int auto_increment primary key,Nom varchar(50),Prixuni float,Categorie varchar(50),img varchar(100),descrip varchar(100));
create table Templates(Id int auto_increment primary key,Img varchar(100),Nom varchar(50),Categorie varchar(50));
create table clienttemplate(Id int primary key,Idcom int,front varchar(50),back varchar(50),template varchar(50),foreign key (Idcom)  references Users(Id));
create table PanierDetailes(Idcom int ,nomproduit varchar(50),typeproduit varchar(50),Quantite int ,taille varchar(50),papier varchar(50),impression varchar(50),coin varchar(50),pellicalage varchar(50),prix double,idtem int,foreign key (idtem)  references  clienttemplate(Id),foreign key (Idcom)  references Users(Id) );
create table Commandes(Id int auto_increment primary key,IdUser int ,DateCommand varchar(50),Statuss varchar(50),FOREIGN KEY (IdUser) REFERENCES Users(Id));
create table detcommandes(idcommande int ,nomproduit varchar(50),typeproduit varchar(50),Quantite int ,taille varchar(50),papier varchar(50),impression varchar(50),coin varchar(50),pellicalage varchar(50),prix double,idtem int,foreign key (idtem)  references clienttemplate(Id));
create table livasion(Id int auto_increment primary key,nom varchar(50),prenom varchar(50),ville varchar(50),adresse varchar(50),tel varchar(50),zip varchar(50),nomcarte varchar(50),numducarte varchar(50),moisexp varchar(50),anneexp varchar(50),cvv varchar(50),Idcommande int, foreign key (Idcommande) references Commandes(Id));
